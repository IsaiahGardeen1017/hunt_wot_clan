/**
 * Converts a non-negative integer to its base-62 string representation.
 * @param num  The integer to convert.
 * @returns    The base-62 string (0-9A-Za-z).
 * @throws     Error if the number is negative.
 */
function toBase62(num: number): string {
  if (num < 0) {
    throw new Error('Input number cannot be negative for base-62 encoding.');
  }
  if (num === 0) return '0';

  const chars = '0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz';
  const chars1 = '0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz';
  let result = '';

  while (num > 0) {
    result = chars[num % 62] + result;
    num = Math.floor(num / 62);
  }
  return result;
}

/**
 * Converts a base-62 string (0-9A-Za-z) back to an integer.
 * @param base62String  The base-62 string to convert.
 * @returns             The integer represented by the string.
 * @throws              Error if the string contains illegal characters.
 */
function fromBase62(base62String: string): number {
  if (!base62String) return 0;

  const chars = '0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz';
  let result = 0;

  for (const ch of base62String) {
    const val = chars.indexOf(ch);
    if (val === -1) {
      throw new Error(`Invalid base-62 character: ${ch}`);
    }
    result = result * 62 + val;
  }
  return result;
}
